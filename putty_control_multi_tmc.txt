static inline int32_t angle_to_xtarget(float angle_deg) {
    return (int32_t)((angle_deg / 360.0f) * COUNTS_PER_REV);
}






uint16_t AS5600_ReadAngleRegister(uint8_t RegAddr)
{
    uint8_t data[2] = {0};
    uint16_t angle = 0;

    // 1. Send the register address to read from (Memory Address)
    HAL_StatusTypeDef status = HAL_I2C_Master_Transmit(
&hi2c1,
        AS5600_ADDRESS,
&RegAddr,
        1,
        100
    );

    if (status != HAL_OK) {
        // Handle error (optional: print error, return 0, etc.)
        return 0;
    }

    // 2. Read two bytes (High Byte and Low Byte)
    status = HAL_I2C_Master_Receive(
&hi2c1,
        AS5600_ADDRESS,
        data,
        2,
        100
    );

    if (status == HAL_OK)
    {
        // Combine the two bytes into a 16-bit value.
        // The AS5600 uses a 12-bit output (0-4095), where the 4 MSBs of the High Byte are zero.
        angle = (uint16_t)((data[0] << 8) | data[1]);
        angle &= 0x0FFF; // Mask to ensure only the 12 relevant bits are kept
    }
    return angle;
}

void uart_print(const char *msg)
{
    if (msg == NULL) return;

    HAL_UART_Transmit(&huart2,
                      (uint8_t *)msg,
                      strlen(msg),
                      HAL_MAX_DELAY);
}

/* -------- SIGNED READ INT -------- */
int32_t tmc_read_int(TMC_Device_t *dev, uint8_t reg)
{
    uint32_t raw = 0;
    tmc5160_read_reg(dev, reg, &raw);
    return (int32_t)raw;
}

/* -------- LOW LEVEL SPI 5 BYTE -------- */
static HAL_StatusTypeDef tmc5160_send40(TMC_Device_t *dev,
                                       uint8_t *tx,
                                       uint8_t *rx)
{
    HAL_StatusTypeDef st;

    TMC_CS_LOW(dev);
    __NOP(); __NOP(); __NOP();

    st = HAL_SPI_TransmitReceive(dev->hspi, tx, rx, 5, 100);

    TMC_CS_HIGH(dev);
    for (volatile int i = 0; i < 200; i++); // ~1 µs CS high time
    return st;
}


/* -------- WRITE REGISTER -------- */
HAL_StatusTypeDef tmc5160_write_reg(TMC_Device_t *dev,uint8_t reg,uint32_t data) {
	uint8_t tx[5], rx[5];
	tx[0] = reg | 0x80;
	tx[1] = data >> 24;
	tx[2] = data >> 16;
	tx[3] = data >> 8;
	tx[4] = data;
	return tmc5160_send40(dev, tx, rx);
}


/* -------- READ REGISTER -------- */
HAL_StatusTypeDef tmc5160_read_reg(TMC_Device_t *dev,uint8_t reg,uint32_t *out) {
	uint8_t tx1[5] = { reg, 0,0,0,0 };
	uint8_t tx2[5] = { 0,0,0,0,0 };
	uint8_t rx[5];
	if (tmc5160_send40(dev, tx1, rx) != HAL_OK)
		return HAL_ERROR;
	if (tmc5160_send40(dev, tx2, rx) != HAL_OK)
		return HAL_ERROR;
	*out = ((uint32_t)rx[1] << 24) |
			((uint32_t)rx[2] << 16) |
			((uint32_t)rx[3] << 8)  |
			((uint32_t)rx[4]);
	return HAL_OK;
}

void tmc_move_all_and_wait(TMC_Device_t **devs,
                           float *angles,
                           uint8_t count)
{
    int done;
    int32_t v;

    /* START MOTION */for (uint8_t i = 0; i < count; i++) {
        TMC_ENABLE(devs[i]);
        tmc5160_write_reg(devs[i],
                          XTARGET_REG,
                          angle_to_xtarget(angles[i]));
    }

    /* WAIT UNTIL ALL MOTORS STOP */do {
        done = 1;
        for (uint8_t i = 0; i < count; i++) {
            v = tmc_read_int(devs[i], VACTUAL_REG);
            if (abs(v) > 5)
                done = 0;
        }
        HAL_Delay(5);
    } while (!done);

    /* DISABLE ALL */for (uint8_t i = 0; i < count; i++)
        TMC_DISABLE(devs[i]);
}
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart){
    if (huart->Instance == USART2)
    {
        if (uart_rx_byte == '\n' || uart_rx_byte == '\r')
        {
            uart_rx_buf[uart_rx_index] = '\0'; // Null-terminate the string
            uart_cmd_ready = 1; // Flag that data is ready to be processed
            uart_rx_index = 0; // Reset index for next command
            }
        else        {
            if (uart_rx_index < UART_RX_BUF_SIZE - 1)
            {
                uart_rx_buf[uart_rx_index++] = uart_rx_byte; // Store received byte
                }
        }
        HAL_UART_Receive_IT(&huart2, &uart_rx_byte, 1); // Wait for next byte
        }
}
uint8_t parse_uart_angles(char *cmd, float *angles, uint8_t max_count)
{
    if (cmd[0] != 'A' || cmd[1] != ':')
        return 0; // Invalid format, must start with 'A:'
    char *token = strtok(&cmd[2], ",");
    uint8_t i = 0;

    while (token && i < max_count)
    {
        angles[i++] = atof(token); // Convert the token to float
        token = strtok(NULL, ","); // Get the next angle token
        }

    // Debug: Print the parsed angles
    for (int i = 0; i < max_count; i++) {
        char dbg[20];
        sprintf(dbg, "Parsed Angle %d: %.2f\r\n", i, angles[i]);
        uart_print(dbg);
    }

    return (i == max_count); // Return true only if the correct number of angles are parsed
    }
/* -------- TMC5160 INIT -------- */
void tmc5160_init(TMC_Device_t *dev){
	TMC_DISABLE(dev);
	HAL_Delay(10);
	tmc5160_write_reg(dev, 0x01, 0x00000007);
	tmc5160_write_reg(dev, 0x90, 0x00061F0A);
	tmc5160_write_reg(dev, 0xEC, 0x000100C3);
	tmc5160_write_reg(dev, 0xA4, 1000);
	tmc5160_write_reg(dev, 0xA5, 50000);
	tmc5160_write_reg(dev, 0xA6, 500);
	tmc5160_write_reg(dev, 0xA7, 200000);
	tmc5160_write_reg(dev, 0xA8, 700);
	tmc5160_write_reg(dev, 0xAA, 1400);
	tmc5160_write_reg(dev, 0xAB, 10);
	tmc5160_write_reg(dev, 0xA0, 0x00000000);
	tmc5160_write_reg(dev, 0xAD, 0);
	HAL_Delay(10);
	// Add status register read after initialization
	uint32_t gstat = 0;
	tmc5160_read_reg(dev, 0x01, &gstat);  // Read the global status register
	char dbg[50];
	    sprintf(dbg, "Motor Status after Init (GSTAT): 0x%08lX\r\n", gstat);
	    uart_print(dbg); // Print the motor's status to verify initialization


	TMC_ENABLE(dev);
}

int main(void)
{

  HAL_Init();
  SystemClock_Config();
  MX_GPIO_Init();
  MX_DMA_Init();
  MX_SPI1_Init();
  MX_USART2_UART_Init();
  MX_I2C1_Init();
  HAL_Delay(50);
  uart_print("UART READY\r\n");
  HAL_UART_Receive_IT(&huart2, &uart_rx_byte, 1);
  TMC_Device_t TMC1 = {
      .hspi = &hspi1,
      .cs_port = GPIOB,
      .cs_pin = GPIO_PIN_6,
      .en_port = GPIOB,
      .en_pin = GPIO_PIN_7,
      .index = 0};

  TMC_Device_t TMC2 = {
      .hspi = &hspi1,
      .cs_port = GPIOC,
      .cs_pin = GPIO_PIN_7,
      .en_port = GPIOB,
      .en_pin = GPIO_PIN_8,
      .index = 1};
  TMC_Device_t *tmc_list[TMC_COUNT] = { &TMC1, &TMC2 };
  uart_print("TMC5160 Init...\r\n");
  uart_print("STM32G474RE AS5600 Detector Initialized.\r\n");
  for (uint8_t i = 0; i < TMC_COUNT; i++)
  {
      tmc5160_init(tmc_list[i]);
      HAL_Delay(20);
  }
  /*
  uint32_t gstat = 0;

  tmc5160_read_reg(&TMC1, 0x01, &gstat);

  char dbg[50];
  sprintf(dbg, "GSTAT = 0x%08lX\r\n", gstat);
  uart_print(dbg);
  */

  HAL_StatusTypeDef status = HAL_I2C_IsDeviceReady(
&hi2c1,             // I2C handle structure (from MX_I2C1_Init)
      AS5600_ADDRESS,     // The 8-bit I2C slave address
      2,                  // Number of trials/retries
      100                 // Timeout in ms
  );
  if (status == HAL_OK)
    {
        // SUCCESS INDICATION: AS5600 is connected and ACKed the address.
	  uart_print("SUCCESS: AS5600 Encoder Detected and Ready!\r\n");
    }
    else
    {
        // FAILURE INDICATION: Device not found or communication failed.
    	uart_print("ERROR: AS5600 Encoder Not Found. Check wiring.\r\n");
    }
  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while (1)
  {
	  if (uart_cmd_ready)
	  {
	      uart_cmd_ready = 0;

	      // Debug: Print the received UART command before parsing
	      uart_print("Received: ");
	      uart_print(uart_rx_buf);  // Print the full received string
	      float angles[TMC_COUNT];
	      if (parse_uart_angles(uart_rx_buf, angles, TMC_COUNT))
	      {
	          uart_print("CMD OK → Moving motors\r\n");
	          tmc_move_all_and_wait(tmc_list, angles, TMC_COUNT);
	          uart_print("Motion done\r\n");
	      }
	      else    {
	          uart_print("CMD ERROR. Format: A:deg0,deg1\r\n");
	      }
	  }
  }


}
