uint16_t AS5600_ReadAngleRegister(uint8_t RegAddr)
{
    uint8_t data[2] = {0};
    uint16_t angle = 0;

    // 1. Send the register address to read from (Memory Address)
    HAL_StatusTypeDef status = HAL_I2C_Master_Transmit(
        &hi2c1,
        AS5600_ADDRESS,
        &RegAddr,
        1,
        100
    );

    if (status != HAL_OK) {
        // Handle error (optional: print error, return 0, etc.)
        return 0;
    }

    // 2. Read two bytes (High Byte and Low Byte)
    status = HAL_I2C_Master_Receive(
        &hi2c1,
        AS5600_ADDRESS,
        data,
        2,
        100
    );

    if (status == HAL_OK)
    {
        // Combine the two bytes into a 16-bit value.
        // The AS5600 uses a 12-bit output (0-4095), where the 4 MSBs of the High Byte are zero.
        angle = (uint16_t)((data[0] << 8) | data[1]);
        angle &= 0x0FFF; // Mask to ensure only the 12 relevant bits are kept
    }
    return angle;
}

/* -------- SIGNED READ INT -------- */
int32_t tmc_read_int(TMC_Device_t *dev, uint8_t reg)
{
    uint32_t raw = 0;
    tmc5160_read_reg(dev, reg, &raw);
    return (int32_t)raw;
}

/* -------- LOW LEVEL SPI 5 BYTE -------- */
static HAL_StatusTypeDef 	(TMC_Device_t *dev, uint8_t *tx, uint8_t *rx) {
	HAL_StatusTypeDef st;
	TMC_CS_LOW(dev);
	__NOP(); __NOP();
	st = HAL_SPI_TransmitReceive(dev->hspi, tx, rx, 5, 100);
	TMC_CS_HIGH(dev);
	return st;
}


/* -------- WRITE REGISTER -------- */
HAL_StatusTypeDef tmc5160_write_reg(TMC_Device_t *dev,uint8_t reg,uint32_t data) {
	uint8_t tx[5], rx[5];
	tx[0] = reg | 0x80;
	tx[1] = data >> 24;
	tx[2] = data >> 16;
	tx[3] = data >> 8;
	tx[4] = data;
	return tmc5160_send40(dev, tx, rx);
}


/* -------- READ REGISTER -------- */
HAL_StatusTypeDef tmc5160_read_reg(TMC_Device_t *dev,uint8_t reg,uint32_t *out) {
	uint8_t tx1[5] = { reg, 0,0,0,0 };
	uint8_t tx2[5] = { 0,0,0,0,0 };
	uint8_t rx[5];
	if (tmc5160_send40(dev, tx1, rx) != HAL_OK)
		return HAL_ERROR;
	if (tmc5160_send40(dev, tx2, rx) != HAL_OK)
		return HAL_ERROR;
	*out = ((uint32_t)rx[1] << 24) |
			((uint32_t)rx[2] << 16) |
			((uint32_t)rx[3] << 8)  |
			((uint32_t)rx[4]);
	return HAL_OK;
}


/* -------- TMC5160 INIT -------- */
void tmc5160_init(TMC_Device_t *dev){
	TMC_DISABLE(dev);
	HAL_Delay(10);
	tmc5160_write_reg(dev, 0x01, 0x00000007);
	tmc5160_write_reg(dev, 0x90, 0x00061F0A);
	tmc5160_write_reg(dev, 0xEC, 0x000100C3);
	tmc5160_write_reg(dev, 0xA4, 1000);
	tmc5160_write_reg(dev, 0xA5, 50000);
	tmc5160_write_reg(dev, 0xA6, 500);
	tmc5160_write_reg(dev, 0xA7, 200000);
	tmc5160_write_reg(dev, 0xA8, 700);
	tmc5160_write_reg(dev, 0xAA, 1400);
	tmc5160_write_reg(dev, 0xAB, 10);
	tmc5160_write_reg(dev, 0xA0, 0x00000000);
	tmc5160_write_reg(dev, 0xAD, 0);
	HAL_Delay(10);
	TMC_ENABLE(dev);
}
