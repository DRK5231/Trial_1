uint16_t AS5600_ReadAngleRegister(uint8_t RegAddr)
{
    uint8_t data[2] = {0};
    uint16_t angle = 0;

    // 1. Send the register address to read from (Memory Address)
    HAL_StatusTypeDef status = HAL_I2C_Master_Transmit(
        &hi2c1,
        AS5600_ADDRESS,
        &RegAddr,
        1,
        100
    );

    if (status != HAL_OK) {
        // Handle error (optional: print error, return 0, etc.)
        return 0;
    }

    // 2. Read two bytes (High Byte and Low Byte)
    status = HAL_I2C_Master_Receive(
        &hi2c1,
        AS5600_ADDRESS,
        data,
        2,
        100
    );

    if (status == HAL_OK)
    {
        // Combine the two bytes into a 16-bit value.
        // The AS5600 uses a 12-bit output (0-4095), where the 4 MSBs of the High Byte are zero.
        angle = (uint16_t)((data[0] << 8) | data[1]);
        angle &= 0x0FFF; // Mask to ensure only the 12 relevant bits are kept
    }
    return angle;
}

void AS5600_WriteRegister(uint8_t reg, uint8_t value)
{
    uint8_t buf[2];
    buf[0] = reg;
    buf[1] = value;
    HAL_I2C_Master_Transmit(
        &hi2c1,
        AS5600_ADDRESS,
        buf,
        2,
        100    );
}

void AS5600_SetZeroPosition(void)
{
    // Read current physical angle
	uint16_t raw = AS5600_ReadAngleRegister(AS5600_REG_RAW_ANGLE_H);// Split 12-bit value into ZPOS registers
	uint8_t zpos_h = (raw >> 8) & 0x0F;  // upper 4 bits
	uint8_t zpos_l = raw & 0xFF;         // lower 8 bits// Write zero position
	AS5600_WriteRegister(AS5600_REG_ZPOS_H, zpos_h);
    AS5600_WriteRegister(AS5600_REG_ZPOS_L, zpos_l);
    HAL_Delay(2); // allow internal update

}

/* -------- SIGNED READ INT -------- */
int32_t tmc_read_int(uint32_t addr)
{
    uint32_t raw = 0;
    tmc5160_read_reg(addr, &raw);
    return (int32_t)raw;  // auto sign-extension
}

/* -------- LOW LEVEL SPI 5 BYTE -------- */
static HAL_StatusTypeDef tmc5160_send40(uint8_t *bytes5, uint8_t *rx5)
{
    HAL_StatusTypeDef st;
    CS_HIGH();
    HAL_Delay(1);
    CS_LOW();
    __NOP(); __NOP();
    st = HAL_SPI_TransmitReceive(&hspi1, bytes5, rx5, 5, 100);
    CS_HIGH();
    HAL_Delay(1);

    return st;
}

/* -------- WRITE REGISTER -------- */
HAL_StatusTypeDef tmc5160_write_reg(uint8_t addr_with_writebit, uint32_t data)
{
    uint8_t tx[5], rx[5];
    tx[0] = addr_with_writebit;
    tx[1] = (data >> 24) & 0xFF;
    tx[2] = (data >> 16) & 0xFF;
    tx[3] = (data >> 8)  & 0xFF;
    tx[4] = data & 0xFF;
    return tmc5160_send40(tx, rx);
}

/* -------- READ REGISTER -------- */
HAL_StatusTypeDef tmc5160_read_reg(uint8_t addr_no_writebit, uint32_t *out)
{
    uint8_t tx1[5] = { addr_no_writebit, 0,0,0,0 };
    uint8_t tx2[5] = { 0,0,0,0,0 };
    uint8_t rx1[5], rx2[5];
    HAL_StatusTypeDef st;
    st = tmc5160_send40(tx1, rx1);
    if (st != HAL_OK) return st;
    HAL_Delay(1);
    st = tmc5160_send40(tx2, rx2);
    if (st != HAL_OK) return st;
    *out = ((uint32_t)rx2[1] << 24) |
           ((uint32_t)rx2[2] << 16) |
           ((uint32_t)rx2[3] << 8)  |
           ((uint32_t)rx2[4]);
    return HAL_OK;
}

/* -------- TMC5160 INIT -------- */
void tmc5160_init(void)
{
    DISABLE_DRV();
    HAL_Delay(10);
    tmc5160_write_reg(0x01, 0x00000007);   // CLEAR GSTAT
    HAL_Delay(2);
    tmc5160_write_reg(0x90, 0x00061F0A);   // IHOLD/IRUN
    HAL_Delay(2);
    tmc5160_write_reg(0xEC, 0x000100c3);   // CHOPCONF
    HAL_Delay(2);
    tmc5160_write_reg(0xA4, 1000);
    tmc5160_write_reg(0xA5, 50000);
    tmc5160_write_reg(0xA6, 500);
    tmc5160_write_reg(0xA7, 200000);
    tmc5160_write_reg(0xA8, 700);
    tmc5160_write_reg(0xAA, 1400);
    tmc5160_write_reg(0xAB, 10);
    tmc5160_write_reg(0xA0, 0x00000000);   // RAMPMODE
    tmc5160_write_reg(0xAD, 51200);           // XTARGET
    HAL_Delay(10);
    ENABLE_DRV();
    HAL_Delay(10);
}
int32_t encoder_read_corrected(void)
{
    int32_t raw = AS5600_ReadAngleRegister(AS5600_REG_RAW_ANGLE_H);

    /* Apply software zero */int32_t val = raw - encoder_zero;

    /* Wrap into 0â€“4095 */if (val < 0) val += 4096;

    /* Direction inversion (KEEP because you need it) */    val = 4095 - val;

    return val;   // 0..4095, PID safe
    }
int32_t encoder_error(int32_t target, int32_t current)
{
    int32_t err = target - current;

    if (err > 2048)  err -= 4096;
    if (err < -2048) err += 4096;

    return err;
}
int32_t pid_compute(int32_t error)
{
    // Deadband handling
    if (abs(error) <= ERROR_DEADBAND)
    {
        pid_integral = 0;
        pid_prev_error = error;
        return 0;
    }

    pid_integral += error;
    float derivative = error - pid_prev_error;

    float out =
        (Kp * error) +
        (Ki * pid_integral) +
        (Kd * derivative);

    // Force minimum correction if error exists
    if (out > 0 && out < PID_MIN_STEP)  out = PID_MIN_STEP;
    if (out < 0 && out > -PID_MIN_STEP) out = -PID_MIN_STEP;

    // Clamp
    if (out > PID_OUT_LIMIT)  out = PID_OUT_LIMIT;
    if (out < -PID_OUT_LIMIT) out = -PID_OUT_LIMIT;

    pid_prev_error = error;

    return (int32_t)out;
}
int main(void)
{


  HAL_Init();
  uint16_t angle_counts = 0;
  SystemClock_Config();
  MX_GPIO_Init();
  MX_DMA_Init();
  MX_SPI1_Init();
  MX_USART2_UART_Init();
  MX_I2C1_Init();
  uart_print("TMC5160 Init...\r\n");
  uart_print("STM32G474RE AS5600 Detector Initialized.\r\n");
  CS_HIGH();
  DISABLE_DRV();
  HAL_Delay(20);
  tmc5160_init();
  tmc5160_write_reg(0xAD, (uint32_t)0);
  HAL_StatusTypeDef status = HAL_I2C_IsDeviceReady(
      &hi2c1,             // I2C handle structure (from MX_I2C1_Init)
      AS5600_ADDRESS,     // The 8-bit I2C slave address
      2,                  // Number of trials/retries
      100                 // Timeout in ms
  );
  if (status == HAL_OK)
    {
        // SUCCESS INDICATION: AS5600 is connected and ACKed the address.
	  uart_print("SUCCESS: AS5600 Encoder Detected and Ready!\r\n");
	  encoder_zero = AS5600_ReadAngleRegister(AS5600_REG_RAW_ANGLE_H);
	  uart_print("AS5600 Software Zero Captured\r\n");
	  //AS5600_SetZeroPosition();
	  //uart_print("AS5600 Zero Position Set\r\n");
        // Turn the Status LED ON solid
    }
    else
    {
        // FAILURE INDICATION: Device not found or communication failed.
    	uart_print("ERROR: AS5600 Encoder Not Found. Check wiring.\r\n");
        // Turn the Status LED OFF
    }
  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  /* ================= USER MAIN LOOP ================= */

  /* ---- TARGET CONFIG (SET ONCE) ---- */
  int32_t target_angle_deg = 90;
  int32_t target_encoder  = (target_angle_deg * ENCODER_CPR) / 360;
  int32_t target_position = (target_angle_deg * MICROSTEPS_PER_REV) / 360;

  while (1)
  {
      char msg[150];

      /* =====================================================
       * OPEN LOOP MOVE (TMC5160 does coarse positioning)
       * ===================================================== */
      uart_print(">>> OPEN LOOP MOVE START <<<\r\n");
      tmc5160_write_reg(XTARGET_REG | 0x80, target_position);

      /* Wait until motor stops */
      while (tmc_read_int(VACTUAL_REG) != 0)
      {
          int32_t enc = encoder_read_corrected();
          float deg = (enc * 360.0f) / ENCODER_CPR;

          sprintf(msg, "ENC=%ld ANG=%.2f\r\n", enc, deg);
          uart_print(msg);

          HAL_Delay(10);
      }

      uart_print(">>> OPEN LOOP MOVE DONE <<<\r\n");

      /* =====================================================
       * PID POSITION TRIM (ENCODER BASED)
       * ===================================================== */
      uart_print(">>> PID TRIM START <<<\r\n");

      while (1)
      {
          int32_t current_encoder = encoder_read_corrected();
          int32_t error = encoder_error(target_encoder, current_encoder);

          int32_t x_actual = tmc_read_int(XACTUAL_REG);
          int32_t pid_step = pid_compute(error);

          int32_t new_xtarget = x_actual + pid_step;
          tmc5160_write_reg(XTARGET_REG | 0x80, new_xtarget);

          sprintf(msg,
              "ENC=%ld ERR=%ld PID=%ld XT=%ld\r\n",
              current_encoder,
              error,
              pid_step,
              new_xtarget
          );
          uart_print(msg);

          if (abs(error) <= ERROR_DEADBAND)
          {
              pid_stable_counter++;

              if (pid_stable_counter >= PID_STABLE_CNT)
              {
                  uart_print(">>> PID LOCKED <<<\r\n");
                  pid_stable_counter = 0;
                  break;
              }
          }else{
              pid_stable_counter = 0;
          }

          HAL_Delay(10);   // 100 Hz PID
      }

      /* =====================================================
       * HOLD POSITION (NO NESTED LOOPS)
       * ===================================================== */
      uart_print(">>> HOLDING POSITION <<<\r\n");
      HAL_Delay(3000);   // Hold torque for 3 seconds

      /* ---- Optional: stop after one cycle ---- */
      // break;
  }
}
